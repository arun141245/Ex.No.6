# **Python Code Generated**

## **1. Code Generated by ChatGPT**

```python
import requests
import json

def fetch_api(url):
    try:
        response = requests.get(url)
        response.raise_for_status()
        return response.json()
    except Exception as e:
        return {"error": str(e)}

weather_url = "https://api.openweathermap.org/data/2.5/weather?q=London&appid=YOUR_KEY"
currency_url = "https://api.exchangerate-api.com/v4/latest/USD"

weather_data = fetch_api(weather_url)
currency_data = fetch_api(currency_url)

print("Weather API Output:")
print(json.dumps(weather_data, indent=4))

print("\nCurrency API Output:")
print(json.dumps(currency_data, indent=4))

# Basic comparison
if "error" in weather_data or "error" in currency_data:
    print("\nSome API returned an error.")
else:
    print("\nBoth APIs returned valid data.")
    print(f"Weather Data Updated At: {weather_data.get('dt', 'N/A')}")
    print(f"Available Currency Rates: {len(currency_data.get('rates', {}))}")
```

## **2. Code Generated by Google Gemini**

```python
import requests

def get_data(url):
    try:
        r = requests.get(url, timeout=5)
        if r.status_code == 200:
            return r.json()
        return {"status": r.status_code, "message": "API Error"}
    except Exception as e:
        return {"message": str(e)}

weather_api = "https://api.openweathermap.org/data/2.5/weather?q=London&appid=YOUR_KEY"
exchange_api = "https://api.exchangerate-api.com/v4/latest/USD"

weather = get_data(weather_api)
exchange = get_data(exchange_api)

print("\n--- WEATHER ---")
print(weather)

print("\n--- EXCHANGE RATES ---")
print(exchange)

# Insights
if "rates" in exchange:
    highest = max(exchange["rates"], key=exchange["rates"].get)
    print(f"\nHighest currency rate: {highest} -> {exchange['rates'][highest]}")

if "main" in weather:
    temp = weather["main"]["temp"]
    print(f"Temperature in London: {temp}")
```
